import java.io.*;
import java.util.Scanner;

public class runLengthEncoding {
	
	public static FileInputStream infile;
	public static FileOutputStream outfile;
	private static int numRow;
	private static int numCol;
	
	public static void methodOne(Scanner scan) {
		PrintStream print = new PrintStream(outfile);
		int row = 0;
		int nextValue = 0;
		int col = 0;
		int count = 0;
		
		boolean found = false;
		
		while (row < numRow) {
			while (col < numCol) {
				if(scan.nextInt() == 0) {
					col++;
					continue;
				} 
				
				int currentValue = scan.nextInt();
				
				if(found == false) {
					print.println(1);
					print.print(row);
					print.print(" ");
					print.print(col);
					print.print(" ");
					//print.print(currentValue);
					print.print(" ");
					col++;
					count++;
					found = true;
					continue;
				}
				if(nextValue == currentValue) {
					count++;
				}
				
				else {
					print.println(count);
					currentValue = nextValue;
					count = 1;
					print.print(row);
					print.print(" ");
					print.print(col);
					print.print(" ");
					print.print(currentValue);
					print.print(" ");
					found = false;
				}
				col++;
			}
			row++;
		}
	}
